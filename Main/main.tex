\begin{frame}[fragile]{Abfrage 1: Gesamtansicht einer Karte}
  \begin{columns}[T,onlytextwidth]
    \column{.57\textwidth}
    \begin{panel}[colframe=accentblue!35]
\begin{lstlisting}
SELECT
  c.card_number,
  c.name,
  t.name AS type,
  c.hp,
  i.name AS illustrator,
  r.name AS rarity,
  GROUP_CONCAT(
    a.name || ' [' || ca.energy_cost || ']',
    ' | '
  ) AS attacks
FROM card c
JOIN type t ON t.id = c.type_id
JOIN illustrator i ON i.id = c.illustrator_id
JOIN rarity r ON r.id = c.rarity_id
LEFT JOIN card_attack ca ON ca.card_id = c.id
LEFT JOIN attack a ON a.id = ca.attack_id
GROUP BY c.id
ORDER BY c.id;
\end{lstlisting}
    \end{panel}
    \column{.39\textwidth}
    \begin{panel}[colframe=accent!35]

\begin{frame}[fragile,shrink=7]{Abfrage 3: Evolutionen per Self Join}
  \begin{columns}[T,onlytextwidth]
    \column{.48\textwidth}
    \begin{panel}[colframe=accentblue!35]
\begin{lstlisting}
SELECT
  c.name AS card,
  e.name AS evolves_from
FROM card c
LEFT JOIN card e
  ON e.id = c.evolves_from_card_id
WHERE c.evolves_from_card_id IS NOT NULL
ORDER BY c.id;
\end{lstlisting}
    \end{panel}
    \column{.48\textwidth}
    \begin{panel}[colframe=accentgold!55!ink]

\begin{frame}[fragile,shrink=6]{Abfrage 4: Karten mit genau zwei Attacken}
  \begin{columns}[T,onlytextwidth]
    \column{.50\textwidth}
    \begin{panel}[colframe=accentblue!35]
\begin{lstlisting}
SELECT
  c.name,
  COUNT(*) AS attacks,
  c.hp
FROM card c
JOIN card_attack ca
  ON ca.card_id = c.id
GROUP BY c.id
HAVING COUNT(*) = 2
ORDER BY c.hp DESC, c.name;
\end{lstlisting}
    \end{panel}
    \column{.46\textwidth}
    \begin{panel}[colframe=accentgold!55!ink]
      {\large\bfseries Trefferliste}\par
      \vspace{0.35em}
      \begin{columns}[T,onlytextwidth]
        \column{.50\textwidth}
        \small
        Ns Gelatwino\\
        Camerupt\\
        Team Rockets Kokowei\\
        Team Rockets Digdri
        \column{.46\textwidth}
        \small
        Ns Gelatroppo\\
        Erikas Ultrigaria\\
        Banette\\
        Ns Flampion
      \end{columns}

      \vspace{0.7em}
      \small
      8 von 21 Karten besitzen genau zwei Attacken. Gerade hier sieht man, warum \texttt{card\_attack} als Zwischentabelle flexibler ist als feste Spalten wie \texttt{attack1} und \texttt{attack2}.
    \end{panel}
  \end{columns}
\end{frame}

\begin{frame}[shrink=8]{Was w\"urden wir beim n\"achsten Mal verbessern?}
  \begin{columns}[T,onlytextwidth]
    \column{.48\textwidth}
    \begin{panel}[colframe=accent!35]
      {\large\bfseries Datenmodell}\par
      \vspace{0.35em}
      \begin{itemize}
        \item Set-Tabelle erg\"anzen statt nur Kartennummern zu speichern
        \item Preis als eigenes Attribut oder Verlaufstabelle erg\"anzen
        \item Alle Evolutionsvorstufen vollst\"andig importieren
        \item Unbenutzte Tabelle \texttt{card\_new} bereinigen oder sinnvoll einsetzen
      \end{itemize}
      \vspace{0.35em}
      \small
      Technik: echte \texttt{FOREIGN KEY}- und \texttt{NOT NULL}-Constraints, Importskript und feste Smoke Tests.
    \end{panel}
    \column{.48\textwidth}
    \begin{panel}[colframe=accentmint!45!ink]
      {\large\bfseries Unser Fazit}\par
      \vspace{0.45em}
      \small
      Die Datenbank ist f\"ur ein Schulprojekt schon gut nutzbar:
      Pflichtdaten sind vollst\"andig, Joins funktionieren und die Kernbeziehungen sind nachvollziehbar modelliert.

      \vspace{0.75em}
      Der n\"achste professionelle Schritt w\"are aber klar:
      \begin{itemize}
        \item mehr Vollst\"andigkeit in den Zusatzdaten
        \item mehr technische Sicherheit im Schema
        \item mehr Automatisierung beim Import
      \end{itemize}

      \vspace{0.35em}
      \centering
      {\Large\bfseries Kleine Datenbasis, aber saubere Idee.}
    \end{panel}
  \end{columns}
\end{frame}
